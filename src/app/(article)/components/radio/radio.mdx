# Radio

Provides single user input from a selection of options.


## Controlled

```tsx
export function App() {
  const [value, setValue] = useState(false);

  return (
    <Unit layout="row">
      <Radio
        checked={value}
        onChange={(event: React.ChangeEvent<HTMLInputElement>) => {
          setValue(event.target.checked);
        }}
      >
        Controlled
      </Radio>
    </Unit>
  );
}
```

## Group

<DemoRadioGroup />

```tsx
const optionsData = [
  {
    name: 'React',
    value: 'react',
  },
  {
    name: 'Vue',
    value: 'vue',
  },
  {
    name: 'Angular',
    value: 'angular',
  },
  {
    name: 'Svelte',
    value: 'svelte',
  },
];

export function App() {
  const [radioGroupValue, setRadioGroupValue] = useState<RadioValue>('angular');

  return (
    <>
      <Radio.Group defaultValue="vue">
        {optionsData.map((item) => (
          <Radio value={item.value} key={item.value}>
            {item.name}
          </Radio>
        ))}
      </Radio.Group>
      <Radio.Group
        value={radioGroupValue}
        onChange={(groupValue) => {
          setRadioGroupValue(groupValue);
        }}
      >
        {optionsData.map((item) => (
          <Radio value={item.value} key={item.value}>
            {item.name}
          </Radio>
        ))}
      </Radio.Group>
      <Radio.Group defaultValue="svelte" disabled>
        {optionsData.map((item) => (
          <Radio value={item.value} key={item.value}>
            {item.name}
          </Radio>
        ))}
      </Radio.Group>
      <Radio.Group defaultValue="vue" layout="column">
        {optionsData.map((item) => (
          <Radio value={item.value} key={item.value}>
            {item.name}
          </Radio>
        ))}
      </Radio.Group>
    </>
  );
}
```

## API

### Radio

| Prop           | Description                                 | Type                                                   | Default |
| -------------- | ------------------------------------------- | ------------------------------------------------------ | ------- |
| defaultChecked | selected or not by default (in single)      | boolean                                                | false   |
| checked        | selected or not (in single)                 | boolean                                                | false   |
| value          | unique identification value (only in group) | string \| number                                       | -       |
| disabled       | disable current radio                       | boolean                                                | false   |
| onChange       | change event handler                        | (event: React.ChangeEvent\<HTMLInputElement\>) => void | -       |
| ...            | native props                                | React.InputHTMLAttributes                              | -       |

### Radio.Group

| Prop         | Description                           | Type                                       | Default |
| ------------ | ------------------------------------- | ------------------------------------------ | ------- |
| defaultValue | default selected items (uncontrolled) | Array\<string \| number\>                  | -       |
| value        | selected items (controlled)           | Array\<string \| number\>                  | -       |
| disabled     | disable all items                     | boolean                                    | false   |
| layout       | vertical layout                       | 'row' \| 'column'                          | 'row'   |
| onChange     | change event handler                  | (value: Array\<string \| number\>) => void | -       |
| ...          | native props                          | React.HTMLAttributes                       | -       |
